pipeline {
    agent any

    stages {
        stage('Clone repository') {
            steps {
                script {
                    // Define your Git repository URL
                    def gitRepoURL = 'https://github.com/Shorya2023/PracticeBDD_Project.git'
                    
                    // Clone the repository
                    git branch: 'master', url: gitRepoURL
                }
            }
        }
        
        stage('Build and test') {
            steps {
                // Run Maven test
                bat 'mvn test'
            }
        }
    }
}
